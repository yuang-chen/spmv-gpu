124  auto get_min_weights = [min_weights, roots, G] __host__ __device__(
125  edge_t const& e // id of edge
126  ) -> bool {
127  // Find the minimum weight for each vertex. If the function returns true,
128  // the weight is the current minimum and the edge will be added to the
129  // output frontier.
130  auto source = G.get_source_vertex(e);
131  auto dest = G.get_destination_vertex(e);
132  auto weight = G.get_edge_weight(e);
133 
134  // If the source and destination are already part of same super vertex, do
135  // not check
136  if (source < dest && roots[source] != roots[dest]) {
137  auto old_weight1 =
138  math::atomic::min(&(min_weights[roots[source]]), weight);
139  auto old_weight2 =
140  math::atomic::min(&(min_weights[roots[dest]]), weight);
141  if (weight <= old_weight1 || weight <= old_weight2) {
142  return true;
143  }
144  }
145  return false;
146  };

148  auto get_min_neighbors =
149  [G, min_weights, min_neighbors, roots] __host__ __device__(
150  edge_t const& e // id of edge
151  ) -> void {
152  // Find the minimum neighbor for each vertex. Use atomic min to break ties
153  // between neighbors that have the same weight.
154  // Consistent ordering (using max here) will prevent loops.
155  // Edges with dest < source are flipped so that reverse edges are treated
156  // as equivalent. Must check that the weight equals the min weight for
157  // that vertex, because some edges can be added to the frontier that are
158  // later beaten by lower weights.
159  auto source = G.get_source_vertex(e);
160  auto dest = G.get_destination_vertex(e);
161  auto weight = G.get_edge_weight(e);
162 
163  if (source < dest && roots[source] != roots[dest]) {
164  if (weight == min_weights[roots[source]]) {
165  math::atomic::min(&(min_neighbors[roots[source]]), e);
166  }
167  if (weight == min_weights[roots[dest]]) {
168  math::atomic::min(&(min_neighbors[roots[dest]]), e);
169  }
170  }
171  };

173  auto add_to_mst = [G, roots, mst_weight, super_vertices, min_neighbors,
174  min_weights, new_roots, not_decremented] __host__
175  __device__(vertex_t const& v) -> void {
176  // Add weights to MST. To prevent duplicate edges, check that either
177  // the source vertex index is
178  // less than the destination vertex index or that the edge with the source
179  // and destination flipped is not included.
180  // Combine super vertices by setting the root of the super vertex to the
181  // root of the destination. Use `roots` to check old roots and update
182  // roots in `new_roots`.
183  if (min_weights[v] != std::numeric_limits<weight_t>::max()) {
184  auto e = min_neighbors[v];
185  auto source = G.get_source_vertex(e);
186  auto dest = G.get_destination_vertex(e);
187  auto weight = G.get_edge_weight(e);
188 
189  if (roots[source] != v) {
190  auto temp = source;
191  source = dest;
192  dest = temp;
193  }
194 
195  if (source < dest || min_neighbors[roots[dest]] != e) {
196  // For large graphs with float weights, there may be slight variance
197  // in the final MST weight due to atomic adds and the amount of
198  // precision loss depending on the order of adds.
199  not_decremented[0] = false;
200  math::atomic::add(&mst_weight[0], weight);
201  math::atomic::add(&super_vertices[0], -1);
202  math::atomic::exch(&new_roots[v], new_roots[dest]);
203  }
204  }
205  };

207  auto jump_pointers_parallel =
208  [new_roots] __host__ __device__(vertex_t const& v) -> void {
209  // Update the root of each vertex. When adding an edge to the MST, we
210  // update the source's root's root to the destination's root.
211  // However, any vertex that had the source's root as its root would not
212  // be updated. We must jump from root to root until the current vertex
213  // and root are equal to find the new roots.
214  vertex_t u = new_roots[v];
215  while (new_roots[u] != u) {
216  u = new_roots[u];
217  }
218  new_roots[v] = u;
219  return;
220  };
