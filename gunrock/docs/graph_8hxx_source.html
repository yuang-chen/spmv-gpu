332 template <typename graph_type>
333 __host__ __device__ double get_average_degree(graph_type const& G) {
334  auto sum = 0;
335  for (auto v = 0; v < G.get_number_of_vertices(); ++v)
336  sum += G.get_number_of_neighbors(v);
337 
338  return (sum / G.get_number_of_vertices());

352 template <typename graph_type>
353 __host__ __device__ double get_degree_standard_deviation(graph_type const& G) {
354  auto average_degree = get_average_degree(G);
355 
356  double accum = 0.0;
357  for (auto v = 0; v < G.get_number_of_vertices(); ++v) {
358  double d = G.get_number_of_neighbors(v);
359  accum += (d - average_degree) * (d - average_degree);
360  }
361  return sqrt(accum / G.get_number_of_vertices());
362 }

377 template <typename graph_type, typename histogram_t>
378 void build_degree_histogram(graph_type const& G,
379  histogram_t* histogram,
380  cuda::stream_t stream = 0) {
381  using vertex_t = typename graph_type::vertex_type;
382  auto length = sizeof(vertex_t) * 8 + 1;
383 
384  // Initialize histogram array to 0s.
385  thrust::fill(thrust::cuda::par.on(stream),
386  histogram + 0, // iterator begin()
387  histogram + length, // iterator end()
388  0 // fill value
389  );
390 
391  // Build the histogram count.
392  auto build_histogram = [=] __device__(vertex_t const& v) {
393  auto degree = G.get_number_of_neighbors(v);
394  vertex_t log_length = 0;
395  while (degree >= (1 << log_length))
396  ++log_length;
397 
398  math::atomic::add(histogram + log_length, (histogram_t)1);
399  };
400 
401  // For each (count from 0...#_of_Vertices), and perform
402  // the operation called build_histogram. Ignore output, as
403  // we are interested in what goes in the pointer histogram.
404  thrust::for_each(thrust::cuda::par.on(stream),
405  thrust::make_counting_iterator<vertex_t>(0), // Begin: 0
406  thrust::make_counting_iterator<vertex_t>(
407  G.get_number_of_vertices()), // End: # of Vertices
408  build_histogram // Unary operation
409  );
410 }
