43 __device__ __host__ coordinates_t mean(graph_t const& G,
44  coordinates_t const* coordinates,
45  std::size_t const& length,
46  vertex_t const& v) {
47  // Calculate mean;
48  float a = 0;
49  float b = 0;
50 
51  auto start_edge = G.get_starting_edge(v);
52  auto num_neighbors = G.get_number_of_neighbors(v);
53 
54  for (auto e = start_edge; e < start_edge + num_neighbors; e++) {
55  vertex_t u = G.get_destination_vertex(e);
56  if (gunrock::util::limits::is_valid(coordinates[u].latitude) &&
57  gunrock::util::limits::is_valid(coordinates[u].longitude)) {
58  // Accumulate the valid latitude and longitude values
59  a += coordinates[u].latitude;
60  b += coordinates[u].longitude;
61  }
62  }
63 
64  a /= length;
65  b /= length;
66 
67  // mean latitude and longitude
68  return {a, b};

74 template <typename vertex_t>
75 __device__ __host__ coordinates_t midpoint(coordinates_t p1,
76  coordinates_t p2,
77  vertex_t const& v) {
78  // Convert to radians
79  p1.latitude = radians(p1.latitude);
80  p1.longitude = radians(p1.longitude);
81  p2.latitude = radians(p2.latitude);
82  p2.longitude = radians(p2.longitude);
83 
84  float bx, by;
85  bx = cos(p2.latitude) * cos(p2.longitude - p1.longitude);
86  by = cos(p2.latitude) * sin(p2.longitude - p1.longitude);
87 
88  coordinates_t mid;
89  mid.latitude =
90  atan2(sin(p1.latitude) + sin(p2.latitude),
91  sqrt((cos(p1.latitude) + bx) * (cos(p1.latitude) + bx) + by * by));
92 
93  mid.longitude = p1.longitude + atan2(by, cos(p1.latitude) + bx);
94 
95  // Convert back to degrees and return the coordinate.
96  mid.latitude = degrees(mid.latitude);
97  mid.longitude = degrees(mid.longitude);
98 
99  return mid;
100 }

106 __device__ __host__ float haversine(coordinates_t n,
107  coordinates_t mean,
108  float radius = 6371) {
109  // Convert degrees to radians
110  n.latitude = radians(n.latitude);
111  n.longitude = radians(n.longitude);
112  mean.latitude = radians(mean.latitude);
113  mean.longitude = radians(mean.longitude);
114 
115  float lat = mean.latitude - n.latitude;
116  float lon = mean.longitude - n.longitude;
117 
118  float a = pow(sin(lat / 2), 2) +
119  cos(n.latitude) * cos(mean.latitude) * pow(sin(lon / 2), 2);
120 
121  float c = 2 * asin(sqrt(a));
122 
123  // haversine distance in km
124  return radius * c;
125 }

138 __device__ __host__ void spatial_median(graph_t& G,
139  std::size_t length,
140  coordinates_t* coordinates,
141  vertex_t v,
142  float* Dinv,
143  int max_iter = 1000,
144  float eps = 1e-3) {
145  float r, rinv;
146  float Dinvs = 0;
147 
148  std::size_t iter_ = 0;
149  std::size_t nonzeros = 0;
150  std::size_t num_zeros = 0;
151 
152  // Can be done cleaner, but storing lat and long.
153  coordinates_t R, tmp, T;
154  coordinates_t y, y1;
155 
156  auto start_edge = G.get_starting_edge(v);
157  auto num_neighbors = G.get_number_of_neighbors(v);
158 
159  // Calculate mean of all <latitude, longitude>
160  // for all possible locations of v;
161  y = mean(G, coordinates, length, v);
162 
163  // Calculate the spatial median;
164  while (true) {
165  iter_ += 1;
166  Dinvs = 0;
167  T.latitude = 0;
168  T.longitude = 0; // T.lat = 0, T.lon = 0;
169  nonzeros = 0;
170 
171  for (auto e = start_edge; e < start_edge + num_neighbors; e++) {
172  vertex_t u = G.get_destination_vertex(e);
173  if (gunrock::util::limits::is_valid(coordinates[u].latitude) &&
174  gunrock::util::limits::is_valid(coordinates[u].longitude)) {
175  // Get the haversine distance between the latitude and longitude of
176  // valid neighbor
177  auto Dist = haversine(coordinates[u], y);
178  Dinv[e] = Dist == 0 ? 0 : 1 / Dist;
179  nonzeros = Dist != 0 ? nonzeros + 1 : nonzeros;
180  Dinvs += Dinv[e];
181  }
182  }
183 
184  std::size_t len = 0;
185  for (auto e = start_edge; e < start_edge + num_neighbors; e++) {
186  // W[] array, Dinv[e] / Dinvs
187  vertex_t u = G.get_destination_vertex(e);
188  if (gunrock::util::limits::is_valid(coordinates[u].latitude) &&
189  gunrock::util::limits::is_valid(coordinates[u].longitude)) {
190  T.latitude += (Dinv[e] / Dinvs) * coordinates[u].latitude;
191  T.longitude += (Dinv[e] / Dinvs) * coordinates[u].longitude;
192  len++;
193  }
194  }
195 
196  num_zeros = len - nonzeros;
197  if (num_zeros == 0) {
198  y1.latitude = T.latitude;
199  y1.longitude = T.longitude;
200  }
201 
202  else if (num_zeros == len) {
203  // Valid location found
204  coordinates[v].latitude = y.latitude;
205  coordinates[v].longitude = y.longitude;
206  return;
207  }
208 
209  else {
210  R.latitude = (T.latitude - y.latitude) * Dinvs;
211  R.longitude = (T.longitude - y.longitude) * Dinvs;
212  r = sqrt(R.latitude * R.latitude + R.longitude * R.longitude);
213 
214  // Was rinv = (r == 0) ?: (num_zeros / r);
215  // https://gcc.gnu.org/onlinedocs/gcc/Conditionals.html
216  // ... I hate myself too.
217  rinv = (r == 0) ? 0 : (num_zeros / r);
218 
219  y1.latitude = max(0.0f, 1 - rinv) * T.latitude +
220  min(1.0f, rinv) * y.latitude; // latitude
221  y1.longitude = max(0.0f, 1 - rinv) * T.longitude +
222  min(1.0f, rinv) * y.longitude; // longitude
223  }
224 
225  tmp.latitude = y.latitude - y1.latitude;
226  tmp.longitude = y.longitude - y1.longitude;
227 
228  if ((sqrt(tmp.latitude * tmp.latitude + tmp.longitude * tmp.longitude)) <
229  eps ||
230  (iter_ > max_iter)) {
231  // Valid location found
232  coordinates[v].latitude = y1.latitude;
233  coordinates[v].longitude = y1.longitude;
234  return;
235  }
236 
237  y.latitude = y1.latitude;
238  y.longitude = y1.longitude;
239  } // -> spatial_median while() loop
240 }
